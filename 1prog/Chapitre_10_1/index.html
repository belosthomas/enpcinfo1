}

	x = X;
	y = Y;
}

Les destructeurs


class obj {
public:
	obj();
	~obj();
	void affiche();
}

obj::obj() {
	cout << "Bonjour !" << endl;
}


obj::~obj() {
	cout << "Au revoir !" << endl;
}

void obj::affiche() {
	cout << "Fonction affiche" << endl;
}

                

int main() {
	obj monobj; // Bonjour !
	monobj.affiche(); // Fonction affiche
	return 0;	// Au revoir !
}

Les destructeurs


class obj {
	char *tab;
	int tabsize;
public:
	obj(int n);
	~obj();
}

obj::obj(int n) {
	cout << "Je construit tab" << endl;
	tab = new char[n]
	tabsize = n;
}

obj::~obj() {
	cout << "Je détruit tab" << endl;
	delete[] tab;
}

                

int main() {
	obj monobj(10);
	obj copiedeobj = monobj;
	return 0; // tab est détruit 2 fois !!
}

Constructeur de copie


class obj {
	char *tab;
	int tabsize;
public:
	obj(int n);
	obj(obj &o);
	~obj();
}

obj::obj(int n) {
	cout << "Je construit tab" << endl;
	tab = new char[n]
	tabsize = n;
}

obj::obj(obj &o) {
	cout << "Je copie tab" << endl;
	tab = new char[o.tabsize];
	tabsize = o.tabsize;
	for (int i = 0; i < tabsize; i++) {
		tab[i] = o.tab[i];
	}
}

obj::~obj() {
	cout << "Je détruit tab" << endl;
	delete[] tab;
}

                

int main() {
	obj monobj(10);
	obj copiedeobj = monobj;
	return 0; // OK !!
}
