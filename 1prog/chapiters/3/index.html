Les variables

On peut voir une variable comme une boîte nommée à l'intérieur de laquelle on place une valeur.

  • Son nom
  • Son type (entier, caractère...)
  • Sa valeur

Définition


int i;

Affectation


i = 2;

Le code s'exécute ligne par ligne


int i;
i = 2;
int j;
j = i;
// Ici i=2 et j=2
i = 1;
// Ici i=2 et j=1

Définition multiple


int k, l, m;

Affectation multiple


k = l = 3;
m = 4;
// Ici k = 3, l = 3 et m = 4

Initialisation


int n=5,o=n,p=INT_MAX;
// Ici n = 5, o = 5 et p = 2147483647

Quelques erreurs à ne pas faire


int q = r = 4; // Erreur !
					

const int s = 12;
s = 13; // Erreur !!

int i;
i=j; // Erreur : j n'existe pas encore

if (j>1) {
	int k=3;
	j=k;
}
i=k; // Erreur: k n'existe plus

Les fonctions

Il faut regrouper les passages identiques en fonctions :

  • Pour obtenir un programme clair...
  • et pour moins se fatiguer !


Attention à bien comprendre quand faire une fonction et à ne pas simplement découper un programme en petits morceaux sans aucune logique.

La base


void afficherCoucou() {
	cout << "Coucou c'est moi !!" << endl;
}

int main() {
	for (int i = 0; i < 10; i++) {
		afficherCoucou();
	}
	return 0;
}

Arguments & Retour


// Nombre entre 0 et n-1
int hasard(int n) {
	return rand() % n;
}

int main() {
	cout << "Génerer un nombre entre 0 et ";
	int max;
	cin >> max;
	cout << "Résultat : " << hasard(max) << endl;
	return 0;
}

Arguments & Retour


// Nombre entre a et b
int hasard(int a, int b) {
	return a+(rand()%(b-a+1));
}

int main() {
	cout << "Minimum : ";
	int min;
	cin >> min;
	cout << "Maximum : ";
	int max;
	cin >> max;
	cout << "Résultat : " << hasard(min, max) << endl;
	return 0;
}

Surcharge


// Nombre entre 0 et n-1

Passage par référence


int echanger(int a, int b) {
	int c = a;
	a = b;
	b = c;
}

int main() {
	int a = 5;
	int b = 10;
	echanger(a, b);
	// Ici a = 5 et b = 10
}

Passage par référence


int echanger(int &a, int &b) {
	int c = a;
	a = b;
	b = c;
}

int main() {
	int a = 5;
	int b = 10;
	echanger(a, b);
	// Ici a = 10 et b = 5
}

Variables locales et globales


int x; // globale

void f() {
	x = 3;
}

	int y; // locale

}
