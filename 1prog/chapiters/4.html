Les tableaux

Définition


// Déclare un tableau de type double de taille 4
double montableau[4];

Affectation


// Assigne à la 1ère case de mon tableau la valeur 8.6;
montableau[0] = 8.6;
// Assigne à la 3ème case mon tableau la valeur 3.4
montableau[2] = 3.4;

Une des erreurs les plus fréquentes en C++


montableau[4] = 5.6; // ERREUR !!

Initialisation


int t[4] = {1, 2, 3, 4};
string s[2] = {"hip", "hop"};

La taille peut être une variable constante


const int n = 5;
string s[n]; // Un tableau de taille 5

Affectation


int t[2];
t = {1, 2}; // Erreur ! Affecter un tableau ne marche pas !

// Il faut traiter les éléments du tableau un par un...
t[0] = 1;
t[1] = 2;
						

int t[2];
int u = {1, 2}; // Erreur ! Affecter un tableau ne marche pas !
t = u; // Ne fait pas ce qu'il faut ! Mais compile sans warning !!

Tableaux et fonctions


void affiche(int s[4]) {
	for (int i = 0; i < 4; i++) {
		cout << s[i] << endl;
	}
}

int main() {
	int t[4] = {1, 2, 3, 4};
	affiche(t);
	return 0;
}

Un tableau est toujours passé par référence


// Rappel: ceci ne marche pas
void affecte1(int x, int val) {
	x = val;
}

// Rappel: c'est ceci qui marche
void affecte2(int &x, int val) {
	x = val;
}

// Une fonctione qui marche sans '&'
void remplit(int s[4], int val) {
	for (int i = 0; i < 4; i++) {
		s[i] = val;
	}
}

Argument de tableau à taille variable


// Une fonctione qui marche sans '&'
void remplit(int s[], int n, int val) {
	for (int i = 0; i < n; i++) {
		s[i] = val;
	}
}

On ne peux pas retourner un tableau


int[4] mafonction() { // Ne fonctionne pas
